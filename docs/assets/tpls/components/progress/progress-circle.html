<div class="cc-progress-circle">
  <strong></strong>
</div>
<script>
  $(function(){
    $('.cc-progress-circle').circleProgress({
      startAngle:-360,
      value: 0.9,
      fill: { gradient: ['#37b4e3', '#1ca8dd'] }
    }).on('circle-animation-progress', function(event, progress,stepValue) {
      $(this).find('strong').html(parseInt(100 * stepValue) + '<i>%</i>');
    });
  })
</script>
<script src="static/lib/js/jquery/circle-progress.js"></script>
