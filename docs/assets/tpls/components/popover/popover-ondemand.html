Add Item 
  <div class="cc-toolbar">
    <div class="cc-toolbar-btns">
      <a href="javascript:void(0)" class="btn cc-btn" role="button" id="add-item" data-toggle="popover" aria-expanded="false">
        <i class="ci ci-add" aria-hidden="true"></i> Add Item</span>
      </a>
    </div>
  </div>
  <script>
  $('#add-item').popover({
    animation:false,
    html: true,
    trigger: 'manual',
    content: function() {
      return $.ajax({url: "sample-add-item.html",
                     dataType: 'html',
                     async: false}).responseText;
    },
    placement:"bottom"
  }).click(function(e) {
    $(this).popover('toggle');
  });
  $('#add-item').on("show.bs.popover",function(){
    var $pop=$(this).data("bs.popover").tip();
    $pop.css({"left":"15px","width":"320px"});
    $pop.addClass("cc-popover");
    $pop.removeAttr("role");
    $pop.find(".arrow").addClass("cc-popover-arrow-l");

  });
  $('#add-item').on("shown.bs.popover",function(){
    var $pop=$(this).data("bs.popover").tip();
    $pop.css({"left":"15px"});
    $pop.attr("aria-expanded",true);
    var $id = $pop.attr("id");
    $(this).removeAttr("aria-describedby");
    $(this).attr("aria-controls",$id);
    $(this).attr("aria-expanded",true);
  });
  $('#add-item').on("hidden.bs.popover",function(){
    var $pop=$(this).data("bs.popover").tip();
    $pop.attr("aria-expanded",false);
    $(this).attr("aria-expanded",false);
  });
  </script>
